coffee, tea,coffee', tea': \integer

0 \leq  coffee \leq 3
\\
0 \leq  tea \leq 3
\\
0 \leq  coffee' \leq 3
\\
0 \leq  tea' \leq 3
\\
tea' = tea
\\
coffee' = coffee - 1
\end{op}
